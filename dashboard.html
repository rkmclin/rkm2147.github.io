  ny_noaa %>% 
  slice_sample(n = 100000) %>%

  mutate(
    year = as.factor(year),
    month = as.factor(month),
    day = as.factor(day),
    tmax = as.integer(tmax),
    tmin = as.integer(tmin)
  ) %>% 
  mutate(
    prcp = (prcp / 10),
    tmax = (tmax / 10),
    tmin = (tmin / 10)
  )

  na.omit() %>%
